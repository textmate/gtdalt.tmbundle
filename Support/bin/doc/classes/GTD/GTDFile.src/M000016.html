    def dump_object
      objects = @actions + @projects + @other_lines + @completed_actions
      objects += @projects.map{|p| EndMarker.new(p.end_line)}
      objects = objects.sort {|a,b| a.line <=> b.line}
      indent = 0
      indent_inc = 2
      s = []
      endA = []
      index = 0
      objects.each do |o|
        case o
        when Project
          t = " " * indent
          s << t + "project #{o.name}"
          indent += indent_inc
        when Action
          t = " " * indent
          t << "@#{o.context} #{o.name}"
          if o.note != nil then
            index += 1
            t << " [#{index}]"
            endA << "[#{index}] #{o.note}"
          end
          if o.due != nil  then
            t << " due:[#{o.due}]"
          end
          s << t
        when Comment
          # t = " " * indent
          s << "##{o.name}"
        when EndMarker
          indent -= indent_inc
          t = " " * indent
          s << t + "end"
        end
      end
      return (s + endA).join("\n")
    end
