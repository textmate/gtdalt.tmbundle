    def initialize(filename)
      @current_project = nil
      @@files << @current_file
      @current_file = filename
      # @current_line = 0
      @projects = Array.new
      @actions = Array.new
      @notes = Hash.new
      @other_lines = Array.new
      @completed_actions = Array.new
      File.open(filename, "r") do |f|
        instructions = GTD::parse(f.read)
        process_instructions(instructions)
      end
    end
