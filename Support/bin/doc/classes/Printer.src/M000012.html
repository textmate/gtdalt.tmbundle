  def to_html
    s = []
    @items.each do |type,item|
      case type
        when :title
          s << "<caption>#{item}</caption>"
        when :table_begin
          s << "<table>"
        when :table_end
          s << "</table>"
        when :row
          s << "<tr>"
          s << item.map{|i| "<td>#{i}</td>"}.join("\n")
          s << "</tr>"
        when :headers
          s << "<tr>"
          s << item.map{|i| "<th>#{i}</th>"}.join("\n")
          s << "</tr>"
        when :raw
          s << item
      end
    end
    return s.join("\n")+"\n"
  end
