  def html_actions_for_context(context)
    # require 'CGI'
    actions = actions_for_context(context)
    ar = []
    ar << "<div>"
    ar << "<h1>Actions for context: #{context}</h1>"
    for name,proj,due_by,location in actions do
      pp due_by
      ar << "<li><a href=\"txmt://open/?url=file://#{location[0].to_s}&line=#{location[1]}\">#{name}</a>\
      in #{proj} #{if due_by != "" then if Date.parse(due_by) < Date.today then 'due by: '+due_by else "<span style=\"color:red\">due by:#{due_by}</span>" end else '' end}</li>"
    end
    ar << "</div>"
    return ar.join("\n")
  end
