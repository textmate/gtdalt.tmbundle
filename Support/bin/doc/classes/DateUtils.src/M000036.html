  def DateUtils::convert_date(string)
    case string.downcase
    when /^today$/
      return Date.today
    when /^tomorrow$/
      return Date.today + 1
    when /^(?:next )?(\w+)$/
      day = Date::ABBR_DAYS[$1] || Date::DAYS[$1]
      raise unless day
      tod = Date.today
      tod=tod.succ while tod.wday != day
      return tod
    when /^(?:in\s*)?((?:[+-])?\d+)\s*(day|week|month|year)(s?)$/
      tod = Date.today
      no = $1.to_i
      case $2
      when "day"
        return tod + no
      when "week"
        return tod + no * 7
      when "month"
        return tod >> no
      when "year"
        return tod + no * 365
      end
    else
      return Date.parse(string.downcase)
    end
  end
